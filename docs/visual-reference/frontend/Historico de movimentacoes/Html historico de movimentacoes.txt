        document.querySelectorAll('tbody tr').forEach(row => {
            row.addEventListener('mouseenter', () => {
                row.querySelector('button').classList.add('scale-105');
            });
            row.addEventListener('mouseleave', () => {
                row.querySelector('button').classList.remove('scale-105');
            });
        });

        // Toggle search bar focus aesthetic
        const searchInput = document.querySelector('input[type="text"]');
        searchInput.addEventListener('focus', () => {
            searchInput.parentElement.classList.add('scale-[1.02]');
        });
        searchInput.addEventListener('blur', () => {
            searchInput.parentElement.classList.remove('scale-[1.02]');
        });
    </script>
</body></html>
